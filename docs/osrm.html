route <- osrmRoute(airports[1, ], airports[2, ], sp = TRUE)

leaflet(width = "100%") %>% addProviderTiles("CartoDB") %>%
  addMarkers(data = airports, popup = ~ name) %>%
  addPolylines(data = route) %>%
  addControl(
    sprintf("<b>Duration</b>: %i min",
